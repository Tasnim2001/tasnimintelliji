    @Override
    public void actionPerformed(@NotNull AnActionEvent e) {

        Project project = e.getProject();
        PsiFile file = e.getData(CommonDataKeys.PSI_FILE);

        if (project == null || file == null) {
            return; // Kein Popup, wie du wolltest
        }

        // Keywords laden
        Map<String, Object> keywords = KeywordLoader.loadKeywords();

        // Scan ausführen → Liste der Matches
        //List<KeywordMatch> results = KeywordScanner.scanFileWithOffsets(file, keywords);
        List<PSISecurityScanner.Result> results = PSISecurityScanner.scan(file, keywords);

        // Ergebnisse als Text aufbereiten
       /* List<String> resultText = results.stream().map(r -> 
                r.main + " → " + r.sub + " | Found: " + r.hit 
        ).toList();*/

        
        List<String> resultText = results.stream().map(r ->
                r.main + " → " + r.sub + " | Found: " + r.hit + " | Line: " + r.line
        ).toList();




        new SecurityReportDialog(
                results.stream().map(r ->
                        r.main + " → " + r.sub + " | Found: " + r.hit
                ).toList()
        ).show();



    }


}
